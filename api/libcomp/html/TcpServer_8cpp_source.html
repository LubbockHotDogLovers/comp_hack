232 DH* TcpServer::GenerateDiffieHellman()
233 {
234  int codes;
235 
236  DH *pDiffieHellman = DH_new();
237 
238  if(nullptr != pDiffieHellman)
239  {
240  if(1 != DH_generate_parameters_ex(pDiffieHellman, DH_KEY_BIT_SIZE,
241  DH_BASE_INT, NULL))
242  {
243  DH_free(pDiffieHellman);
244  pDiffieHellman = nullptr;
245  }
246 
247  const BIGNUM *p = nullptr, *q = nullptr, *g = nullptr;
248 
249  if(pDiffieHellman)
250  {
251  DH_get0_pqg(pDiffieHellman, &p, &q, &g);
252  }
253 
254  if(pDiffieHellman && (nullptr == p || nullptr == g ||
255  1 != DH_check(pDiffieHellman, &codes) ||
256  DH_SHARED_DATA_SIZE != DH_size(pDiffieHellman)))
257  {
258  DH_free(pDiffieHellman);
259  pDiffieHellman = nullptr;
260  }
261  }
