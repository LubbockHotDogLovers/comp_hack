391  bool LoadObjectsFromFile(gsl::not_null<DataStore*> pDataStore,
392  const libcomp::String& filePath,
393  DefinitionManager* definitionManager = nullptr)
394  {
395  tinyxml2::XMLDocument objsDoc;
396 
397  std::vector<char> data = pDataStore->ReadFile(filePath);
398 
399  if(data.empty())
400  {
401  LOG_WARNING(libcomp::String("File does not exist or is"
402  " empty: %1\n").Arg(filePath));
403  return true;
404  }
405 
406  if(tinyxml2::XML_SUCCESS !=
407  objsDoc.Parse(&data[0], data.size()))
408  {
409  return false;
410  }
411 
412  const tinyxml2::XMLElement *rootNode = objsDoc.RootElement();
413  const tinyxml2::XMLElement *objNode = rootNode->FirstChildElement("object");
414 
415  while(nullptr != objNode)
416  {
417  if(!LoadObject<T>(objsDoc, objNode, definitionManager))
418  {
419  LOG_ERROR(libcomp::String("Failed to load XML file: %1\n")
420  .Arg(filePath));
421  return false;
422  }
423 
424  objNode = objNode->NextSiblingElement("object");
425  }
426 
427  LOG_DEBUG(libcomp::String("Loaded XML file: %1\n").Arg(filePath));
428 
429  return true;
